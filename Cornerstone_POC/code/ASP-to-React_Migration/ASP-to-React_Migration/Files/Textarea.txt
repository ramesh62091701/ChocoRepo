import { TextField } from "@flare/primitives/textbox";
import { DetailsState } from "actions/SelectDetailsActions";
import React from "react";
import { useLocalizationsDefaults } from "@flare/primitives/core-ui";
import { TrainingDetails } from "localizations";

interface Props {
    inputComment: string | undefined;
    setInputComment: (
        field: keyof DetailsState,
        value: string | Date | null
    ) => void;
}

const Comments: React.FC<Props> = ({ inputComment, setInputComment }) => {
    const CommentsLabel = useLocalizationsDefaults(`${TrainingDetails.Comments}`);
    return (
        <TextField
            errorMessage="Please enter valid details"
            id="adaptive-width-textfield"
            label={CommentsLabel}
            labelSize="small"
            placeholder={CommentsLabel}
            size="medium"
            onChange={(comment) => {
                setInputComment("comments", comment.target.value);
            }}
            value={inputComment}
        />
    );
};

export default Comments;
